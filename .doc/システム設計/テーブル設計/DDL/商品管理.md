-- 60-Items
CREATE TABLE items (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT '商品のユニークなID',
    item_code VARCHAR(64) NOT NULL UNIQUE COMMENT '商品コード（ユニーク）',
    site_id INT NOT NULL COMMENT '商品が所属するサイトID',
    category_id INT NOT NULL COMMENT '商品カテゴリID',
    maker_name VARCHAR(64) DEFAULT NULL COMMENT 'メーカー名',
    name VARCHAR(64) NOT NULL COMMENT '商品名',
    description TEXT DEFAULT NULL COMMENT '商品の説明',
    unit_price DECIMAL(10,2) NOT NULL COMMENT '商品の単価',
    unit_id INT NOT NULL COMMENT '商品単位ID',
    from_source ENUM('MANUAL', 'IMPORT') NOT NULL COMMENT '商品登録のソース（手動/インポート）',
    is_recommended BOOLEAN DEFAULT FALSE COMMENT '推奨商品かどうか',
    published_at DATETIME DEFAULT NULL COMMENT '公開日時',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    FOREIGN KEY (site_id) REFERENCES sites(id) ON DELETE CASCADE,
    FOREIGN KEY (category_id) REFERENCES item_categories(id) ON DELETE SET NULL,
    FOREIGN KEY (unit_id) REFERENCES item_units(id) ON DELETE SET NULL
) COMMENT='商品情報を管理するテーブル';

-- 61-ItemCategories
CREATE TABLE item_categories (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT '商品カテゴリのユニークなID',
    category_code VARCHAR(64) NOT NULL UNIQUE COMMENT 'カテゴリコード（ユニーク）',
    site_id INT NOT NULL COMMENT '商品カテゴリが所属するサイトID',
    name VARCHAR(64) NOT NULL COMMENT '商品カテゴリ名',
    priority INT NOT NULL DEFAULT 1 COMMENT 'カテゴリの優先順位',
    is_published BOOLEAN DEFAULT FALSE COMMENT '公開状態（デフォルトは非公開）',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    FOREIGN KEY (site_id) REFERENCES sites(id) ON DELETE CASCADE
) COMMENT='商品カテゴリ情報を管理するテーブル';

-- 62-ItemUnits
CREATE TABLE item_units (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT '商品単位のユニークなID',
    unit_code VARCHAR(64) NOT NULL UNIQUE COMMENT '商品単位コード（ユニーク）',
    site_id INT NOT NULL COMMENT '商品単位が所属するサイトID',
    name VARCHAR(64) NOT NULL COMMENT '商品単位名',
    priority INT NOT NULL DEFAULT 1 COMMENT '単位の優先順位',
    is_published BOOLEAN DEFAULT FALSE COMMENT '公開状態（デフォルトは非公開）',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    FOREIGN KEY (site_id) REFERENCES sites(id) ON DELETE CASCADE
) COMMENT='商品単位情報を管理するテーブル';

-- 63-ItemStocks
CREATE TABLE item_stocks (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT '在庫のユニークなID',
    item_id INT NOT NULL COMMENT '商品ID',
    site_id INT NOT NULL COMMENT 'サイトID',
    is_infinite_stock BOOLEAN NOT NULL DEFAULT FALSE COMMENT '無限大在庫フラグ',
    stock_quantity INT DEFAULT 0 COMMENT '在庫数',
    safety_stock_quantity INT DEFAULT 0 COMMENT '安全在庫数',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    FOREIGN KEY (item_id) REFERENCES items(id) ON DELETE CASCADE,
    FOREIGN KEY (site_id) REFERENCES sites(id) ON DELETE CASCADE,
    UNIQUE INDEX uq_item_site (item_id, site_id)
) COMMENT='商品在庫情報を管理するテーブル';
