-- LoginAttempts
CREATE TABLE login_attempts (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'ログイン試行のユニークなID',
    login_id VARCHAR(64) NOT NULL COMMENT '試行されたログインID',
    ip_address VARCHAR(39) NOT NULL COMMENT 'ログイン試行を行ったIPアドレス',
    is_success BOOLEAN NOT NULL DEFAULT FALSE COMMENT '成功フラグ（デフォルトは失敗）',
    failure_reason TEXT DEFAULT NULL COMMENT '失敗理由（成功時はNULL）',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '試行日時'
) COMMENT='ログイン試行情報を管理するテーブル';
