-- -----------------------------------
-- お知らせ管理 (Notification Management)
-- version 1.0
-- -----------------------------------

-- 30-Notifications
CREATE TABLE notifications (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'お知らせのユニークなID',
    notification_code VARCHAR(64) NOT NULL UNIQUE COMMENT 'お知らせコード（ユニーク）',
    category_id INT NOT NULL COMMENT 'お知らせカテゴリID',
    title VARCHAR(64) NOT NULL COMMENT 'お知らせタイトル',
    content TEXT NOT NULL COMMENT 'お知らせ内容',
    publish_start_at DATETIME NOT NULL COMMENT '公開開始日時',
    publish_end_at DATETIME DEFAULT NULL COMMENT '公開終了日時',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    FOREIGN KEY (category_id) REFERENCES notification_categories(id) ON DELETE CASCADE
) COMMENT='お知らせ情報を管理するテーブル';

-- 31-NotificationCategories
CREATE TABLE notification_categories (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'カテゴリのユニークなID',
    name VARCHAR(64) NOT NULL COMMENT 'カテゴリ名',
    description VARCHAR(255) DEFAULT NULL COMMENT 'カテゴリ説明',
    parent_id INT DEFAULT NULL COMMENT '親カテゴリID（階層構造用）',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    FOREIGN KEY (parent_id) REFERENCES notification_categories(id) ON DELETE SET NULL
) COMMENT='お知らせカテゴリ情報を管理するテーブル';

-- 32-NotificationReceivers
CREATE TABLE notification_receivers (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT '受信対象のユニークなID',
    notification_id INT NOT NULL COMMENT 'お知らせID',
    entity_type VARCHAR(64) NOT NULL COMMENT '受信対象のエンティティ種別（例: Users, Operators）',
    entity_id INT NOT NULL COMMENT '受信対象のエンティティID',
    send_method_id INT NOT NULL COMMENT '送信方法ID',
    sent_at DATETIME NOT NULL COMMENT '送信日時',
    is_read BOOLEAN NOT NULL DEFAULT FALSE COMMENT '既読フラグ（デフォルトは未読）',
    read_at DATETIME DEFAULT NULL COMMENT '既読日時',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    FOREIGN KEY (notification_id) REFERENCES notifications(id) ON DELETE CASCADE,
    FOREIGN KEY (send_method_id) REFERENCES notification_send_methods(id) ON DELETE CASCADE
) COMMENT='お知らせ受信対象を管理するテーブル';

-- 33-NotificationSenders
CREATE TABLE notification_senders (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT '送信者のユニークなID',
    notification_id INT NOT NULL COMMENT 'お知らせID',
    entity_type VARCHAR(64) NOT NULL COMMENT '送信者のエンティティ種別',
    entity_id INT NOT NULL COMMENT '送信者のエンティティID',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    FOREIGN KEY (notification_id) REFERENCES notifications(id) ON DELETE CASCADE
) COMMENT='お知らせ送信者を管理するテーブル';

-- 34-NotificationSendMethods
CREATE TABLE notification_send_methods (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT '送信方法のユニークなID',
    name VARCHAR(64) NOT NULL COMMENT '送信方法名',
    description VARCHAR(255) DEFAULT NULL COMMENT '送信方法の説明',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）'
) COMMENT='お知らせ送信方法を管理するテーブル';
