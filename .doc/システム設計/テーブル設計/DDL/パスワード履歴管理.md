-- PasswordHistories
CREATE TABLE password_histories (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'パスワード履歴のユニークなID',
    authenticate_id INT NOT NULL COMMENT '関連する認証ID',
    password VARCHAR(255) NOT NULL COMMENT '保存されたパスワード（ハッシュ化）',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT 'パスワード保存日時',
    FOREIGN KEY (authenticate_id) REFERENCES authenticates(id) ON DELETE CASCADE
) COMMENT='パスワード履歴情報を管理するテーブル';
