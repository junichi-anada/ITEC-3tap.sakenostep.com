-- UserExternalCodes
CREATE TABLE user_external_codes (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'ユーザー外部コードのユニークなID',
    user_id INT NOT NULL COMMENT '関連するユーザーID',
    external_code VARCHAR(255) NOT NULL COMMENT '外部システムでのコード',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
) COMMENT='ユーザーの外部システムコードを管理するテーブル';
