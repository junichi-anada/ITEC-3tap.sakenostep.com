%% -----------------------------------
%% 注文管理 (Order Management)
%% version 1.0
%% -----------------------------------
erDiagram
  Sites ||--o{ Orders : "belong"
  Users ||--o{ Orders : "belong"
  Orders ||--o{ OrderDetails : "belong"
  OrderDetails ||--|| Items : "has"

  %% 70-Orders
  %% 注文情報を管理するテーブル
  Orders {
    integer id PK "AUTO_INCREMENT" %% 注文のユニークなID
    string(64) order_code UK "NOT_NULL" %% 注文コード（ユニーク）
    reference site_id FK "NOT_NULL: Sites.id" %% 注文が発生したサイトID
    reference user_id FK "NOT_NULL: Users.id" %% 注文を行ったユーザーID
    decimal total_price "(10,2)" %% 合計金額
    decimal tax "(8,0)" %% 税額
    decimal postage "(4,0)" %% 送料
    datetime ordered_at %% 注文日時
    datetime processed_at %% 処理日時
    datetime exported_at %% 出力日時
    datetime created_at "NOT_NULL" %% 作成日時
    datetime updated_at %% 更新日時
    datetime deleted_at %% 削除日時（論理削除用）
  }

  %% 71-OrderDetails
  %% 注文明細を管理するテーブル
  OrderDetails {
    integer id PK "AUTO_INCREMENT" %% 注文明細のユニークなID
    string(64) detail_code UK "NOT_NULL" %% 明細コード（ユニーク）
    reference order_id FK "NOT_NULL: Orders.id" %% 関連する注文ID
    reference item_id FK "NOT_NULL: Items.id" %% 注文された商品のID
    integer volume "NOT_NULL" %% 注文数量
    decimal unit_price "NOT_NULL: (10,2)" %% 商品の単価
    string(64) unit_name "NOT_NULL" %% 商品単位名
    decimal price "NOT_NULL: (10,2)" %% 合計金額（数量 × 単価）
    decimal tax "NOT_NULL: (4,0)" %% 税額
    datetime processed_at %% 処理日時
    datetime created_at %% 作成日時
    datetime updated_at %% 更新日時
    datetime deleted_at %% 削除日時（論理削除用）
  }
