-- 70-Orders
CREATE TABLE orders (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT '注文のユニークなID',
    order_code VARCHAR(64) NOT NULL UNIQUE COMMENT '注文コード（ユニーク）',
    site_id INT NOT NULL COMMENT '注文が発生したサイトID',
    user_id INT NOT NULL COMMENT '注文を行ったユーザーID',
    total_price DECIMAL(10,2) DEFAULT NULL COMMENT '合計金額',
    tax DECIMAL(8,0) DEFAULT NULL COMMENT '税額',
    postage DECIMAL(4,0) DEFAULT NULL COMMENT '送料',
    ordered_at DATETIME DEFAULT NULL COMMENT '注文日時',
    processed_at DATETIME DEFAULT NULL COMMENT '処理日時',
    exported_at DATETIME DEFAULT NULL COMMENT '出力日時',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    FOREIGN KEY (site_id) REFERENCES sites(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
) COMMENT='注文情報を管理するテーブル';

-- 71-OrderDetails
CREATE TABLE order_details (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT '注文明細のユニークなID',
    detail_code VARCHAR(64) NOT NULL UNIQUE COMMENT '明細コード（ユニーク）',
    order_id INT NOT NULL COMMENT '関連する注文ID',
    item_id INT NOT NULL COMMENT '注文された商品のID',
    volume INT NOT NULL COMMENT '注文数量',
    unit_price DECIMAL(10,2) NOT NULL COMMENT '商品の単価',
    unit_name VARCHAR(64) NOT NULL COMMENT '商品単位名',
    price DECIMAL(10,2) NOT NULL COMMENT '合計金額（数量 × 単価）',
    tax DECIMAL(4,0) NOT NULL COMMENT '税額',
    processed_at DATETIME DEFAULT NULL COMMENT '処理日時',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    FOREIGN KEY (order_id) REFERENCES orders(id) ON DELETE CASCADE,
    FOREIGN KEY (item_id) REFERENCES items(id) ON DELETE CASCADE
) COMMENT='注文明細を管理するテーブル';
