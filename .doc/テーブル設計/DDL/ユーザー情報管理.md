-- 80-Users
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT '顧客のユニークなID',
    user_code VARCHAR(64) NOT NULL UNIQUE COMMENT '顧客コード（ユニーク）',
    site_id INT NOT NULL COMMENT '顧客が所属するサイトID',
    name VARCHAR(64) NOT NULL COMMENT '顧客名',
    last_name VARCHAR(32) DEFAULT NULL COMMENT '姓',
    first_name VARCHAR(32) DEFAULT NULL COMMENT '名',
    last_name_kana VARCHAR(32) DEFAULT NULL COMMENT '姓（カナ）',
    first_name_kana VARCHAR(32) DEFAULT NULL COMMENT '名（カナ）',
    postal_code VARCHAR(10) DEFAULT NULL COMMENT '郵便番号',
    address VARCHAR(128) DEFAULT NULL COMMENT '住所',
    phone VARCHAR(24) DEFAULT NULL COMMENT '電話番号',
    phone2 VARCHAR(24) DEFAULT NULL COMMENT '補助電話番号',
    fax VARCHAR(24) DEFAULT NULL COMMENT 'FAX番号',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    FOREIGN KEY (site_id) REFERENCES sites(id) ON DELETE CASCADE
) COMMENT='ユーザー情報を管理するテーブル';
