-- SecurityLogs
CREATE TABLE security_logs (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'セキュリティログのユニークなID',
    entity_type VARCHAR(64) NOT NULL COMMENT '操作エンティティの種別（例: User, Operator, Company, Admin）',
    entity_id INT NOT NULL COMMENT '操作を行ったエンティティID',
    action_type VARCHAR(32) NOT NULL COMMENT '操作の種類（例: LOGIN, LOGOUT, DATA_ACCESS, DATA_CREATE, DATA_UPDATE, DATA_DELETE）',
    action_detail VARCHAR(255) DEFAULT NULL COMMENT '操作の詳細',
    ip_address VARCHAR(39) NOT NULL COMMENT '操作者のIPアドレス',
    user_agent TEXT DEFAULT NULL COMMENT '操作者のユーザーエージェント',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '操作日時'
) COMMENT='セキュリティログ情報を管理するテーブル';
