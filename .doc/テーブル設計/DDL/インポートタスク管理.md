-- ImportTasks
CREATE TABLE import_tasks (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'インポートタスクのユニークなID',
    task_code VARCHAR(64) NOT NULL UNIQUE COMMENT 'タスクコード（ユニーク）',
    site_id INT NOT NULL COMMENT '関連するサイトID',
    data_type VARCHAR(64) NOT NULL COMMENT 'インポートするデータの種類',
    file_path VARCHAR(255) NOT NULL COMMENT 'インポートするファイルのパス',
    status ENUM('Pending', 'Processing', 'Completed') NOT NULL COMMENT 'タスクのステータス',
    status_message TEXT DEFAULT NULL COMMENT 'ステータスメッセージ',
    operator_id INT NOT NULL COMMENT 'タスク実行者（オペレータID）',
    uploaded_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT 'アップロード日時',
    imported_at DATETIME DEFAULT NULL COMMENT 'インポート完了日時',
    FOREIGN KEY (site_id) REFERENCES sites(id) ON DELETE CASCADE,
    FOREIGN KEY (operator_id) REFERENCES operators(id) ON DELETE CASCADE
) COMMENT='インポートタスク情報を管理するテーブル';
