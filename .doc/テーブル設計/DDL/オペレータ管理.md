-- 20-Operators
CREATE TABLE operators (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'オペレータのユニークなID',
    operator_code VARCHAR(64) NOT NULL UNIQUE COMMENT 'オペレータコード（ユニーク）',
    company_id INT NOT NULL COMMENT '所属する会社ID',
    name VARCHAR(32) NOT NULL COMMENT 'オペレータ名',
    operator_rank_id INT NOT NULL COMMENT 'オペレータランクID',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    INDEX idx_operators_company_id (company_id),
    INDEX idx_operators_operator_rank_id (operator_rank_id),
    FOREIGN KEY (company_id) REFERENCES companies(id) ON DELETE CASCADE,
    FOREIGN KEY (operator_rank_id) REFERENCES operator_ranks(id) ON DELETE CASCADE
) COMMENT='各オペレータの情報を管理するテーブル';

-- 21-SiteOperators
CREATE TABLE site_operators (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'サイトオペレータのユニークなID',
    site_id INT NOT NULL COMMENT '関連するサイトID',
    operator_id INT NOT NULL COMMENT '関連するオペレータID',
    role_id INT NOT NULL COMMENT 'サイトオペレータの役割ID',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    UNIQUE INDEX uq_site_operators (site_id, operator_id, role_id),
    FOREIGN KEY (site_id) REFERENCES sites(id) ON DELETE CASCADE,
    FOREIGN KEY (operator_id) REFERENCES operators(id) ON DELETE CASCADE,
    FOREIGN KEY (role_id) REFERENCES site_operator_roles(id) ON DELETE CASCADE
) COMMENT='サイトに紐づくオペレータを管理するテーブル';

-- 22-OperatorRanks
CREATE TABLE operator_ranks (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'オペレータランクのユニークなID',
    name VARCHAR(32) NOT NULL COMMENT 'ランク名',
    priority INT NOT NULL DEFAULT 1 COMMENT 'ランクの優先度',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）'
) COMMENT='オペレータランクを管理するテーブル';

-- 23-SiteOperatorRoles
CREATE TABLE site_operator_roles (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'サイトオペレータロールのユニークなID',
    name VARCHAR(32) NOT NULL COMMENT '役割名',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）'
) COMMENT='サイトオペレータの役割を管理するテーブル';

-- 24-Companies
CREATE TABLE companies (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT '会社のユニークなID',
    company_code VARCHAR(64) NOT NULL UNIQUE COMMENT '会社コード（ユニーク）',
    company_name VARCHAR(32) NOT NULL COMMENT '会社名',
    name VARCHAR(64) NOT NULL COMMENT '登録者名または代表者名',
    postal_code VARCHAR(10) DEFAULT NULL COMMENT '郵便番号',
    address VARCHAR(128) DEFAULT NULL COMMENT '住所',
    phone VARCHAR(24) DEFAULT NULL COMMENT '電話番号',
    phone2 VARCHAR(24) DEFAULT NULL COMMENT '補助電話番号',
    fax VARCHAR(24) DEFAULT NULL COMMENT 'FAX番号',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）'
) COMMENT='会社情報を管理するテーブル';

-- 25-Sites
CREATE TABLE sites (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'サイトのユニークなID',
    site_code VARCHAR(64) NOT NULL UNIQUE COMMENT 'サイトコード（ユニーク）',
    company_id INT NOT NULL COMMENT '所属する会社ID',
    url VARCHAR(255) NOT NULL UNIQUE COMMENT 'サイトのURL',
    name VARCHAR(64) NOT NULL COMMENT 'サイト名',
    description VARCHAR(255) DEFAULT NULL COMMENT 'サイトの説明',
    is_btob BOOLEAN NOT NULL DEFAULT FALSE COMMENT 'BtoBサイトかどうか',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    FOREIGN KEY (company_id) REFERENCES companies(id) ON DELETE CASCADE
) COMMENT='サイト情報を管理するテーブル';
