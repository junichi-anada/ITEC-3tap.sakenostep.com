-- 10-Authenticates
CREATE TABLE authenticates (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'ユニークな認証レコードID',
    auth_code VARCHAR(64) NOT NULL UNIQUE COMMENT '認証コード（ユニーク）',
    site_id INT NOT NULL COMMENT '関連するサイトID',
    entity_type VARCHAR(64) NOT NULL COMMENT '認証対象のエンティティ種別（例: Users, Companies）',
    entity_id INT NOT NULL COMMENT '認証対象のエンティティID',
    login_code VARCHAR(50) NOT NULL COMMENT 'ログインに使用するコード',
    password VARCHAR(255) NOT NULL COMMENT 'ハッシュ化されたパスワード',
    expires_at DATETIME NOT NULL COMMENT '認証の有効期限',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    INDEX idx_authenticates_login_code (login_code),
    INDEX idx_authenticates_password (password),
    FOREIGN KEY (site_id) REFERENCES sites(id) ON DELETE CASCADE
) COMMENT='認証情報を管理するテーブル';

-- 11-AuthenticateOauths
CREATE TABLE authenticate_oauths (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'ユニークなOAuth認証レコードID',
    auth_code VARCHAR(64) NOT NULL UNIQUE COMMENT 'OAuth認証コード（ユニーク）',
    site_id INT NOT NULL COMMENT '関連するサイトID',
    entity_type VARCHAR(64) NOT NULL COMMENT '認証対象のエンティティ種別',
    entity_id INT NOT NULL COMMENT '認証対象のエンティティID',
    auth_provider_id INT NOT NULL COMMENT '使用した認証プロバイダーID',
    token VARCHAR(255) NOT NULL COMMENT 'ハッシュ化されたトークン',
    expires_at DATETIME NOT NULL COMMENT 'トークンの有効期限',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    INDEX idx_authenticate_oauths_entity (entity_type, entity_id),
    FOREIGN KEY (site_id) REFERENCES sites(id) ON DELETE CASCADE,
    FOREIGN KEY (auth_provider_id) REFERENCES auth_providers(id) ON DELETE CASCADE
) COMMENT='OAuth認証情報を管理するテーブル';

-- 12-AuthProviders
CREATE TABLE auth_providers (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT '認証プロバイダーのユニークなID',
    provider_code VARCHAR(64) NOT NULL UNIQUE COMMENT 'プロバイダーコード（ユニーク）',
    name VARCHAR(32) NOT NULL COMMENT '認証プロバイダー名',
    description VARCHAR(255) DEFAULT NULL COMMENT 'プロバイダーの説明',
    is_enable BOOLEAN NOT NULL DEFAULT FALSE COMMENT '利用可能かどうか（デフォルトは無効）',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）'
) COMMENT='認証プロバイダー情報を管理するテーブル';

-- 13-UsableSites
CREATE TABLE usable_sites (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'ユニークな使用可能サイトのレコードID',
    entity_type VARCHAR(64) NOT NULL COMMENT 'サイト利用対象のエンティティ種別',
    entity_id INT NOT NULL COMMENT 'サイト利用対象のエンティティID',
    site_id INT NOT NULL COMMENT '利用可能なサイトID',
    shared_login_allowed BOOLEAN NOT NULL DEFAULT FALSE COMMENT '共有ログインの可否（デフォルトは不可）',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    INDEX idx_usable_sites_entity (entity_type, entity_id),
    FOREIGN KEY (site_id) REFERENCES sites(id) ON DELETE CASCADE
) COMMENT='使用可能なサイト情報を管理するテーブル';

-- 14-SiteAuthProviders
CREATE TABLE site_auth_providers (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'サイトで使用可能な認証プロバイダーのレコードID',
    site_id INT NOT NULL COMMENT '関連するサイトID',
    auth_provider_id INT NOT NULL COMMENT '使用可能な認証プロバイダーID',
    is_enabled BOOLEAN NOT NULL DEFAULT TRUE COMMENT '利用可能かどうか（デフォルトは有効）',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新日時',
    UNIQUE INDEX uq_site_auth_providers (site_id, auth_provider_id),
    FOREIGN KEY (site_id) REFERENCES sites(id) ON DELETE CASCADE,
    FOREIGN KEY (auth_provider_id) REFERENCES auth_providers(id) ON DELETE CASCADE
) COMMENT='サイトごとに利用可能な認証プロバイダーを管理するテーブル';
