-- ImportExportLogs
CREATE TABLE import_export_logs (
    id INT AUTO_INCREMENT PRIMARY KEY COMMENT 'ログのユニークなID',
    log_code VARCHAR(64) NOT NULL UNIQUE COMMENT 'ログコード（ユニーク）',
    site_id INT NOT NULL COMMENT '関連するサイトID',
    operator_id INT NOT NULL COMMENT '操作を実行したオペレータID',
    operation_type ENUM('IMPORT', 'EXPORT') NOT NULL COMMENT '操作の種類（インポート/エクスポート）',
    target_type ENUM('USER', 'ITEM', 'ORDER') NOT NULL COMMENT '対象データの種類',
    file_name VARCHAR(255) NOT NULL COMMENT 'ファイル名',
    record_count INT NOT NULL COMMENT '対象レコード数',
    status ENUM('SUCCESS', 'FAILED') NOT NULL COMMENT '操作結果のステータス',
    error_detail TEXT DEFAULT NULL COMMENT 'エラー詳細（失敗時）',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '作成日時',
    deleted_at DATETIME DEFAULT NULL COMMENT '削除日時（論理削除用）',
    FOREIGN KEY (site_id) REFERENCES sites(id) ON DELETE CASCADE,
    FOREIGN KEY (operator_id) REFERENCES operators(id) ON DELETE SET NULL
) COMMENT='インポート/エクスポートログ情報を管理するテーブル';
